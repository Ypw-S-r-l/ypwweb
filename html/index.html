YPW Account


import http.client
import json

conn = http.client.HTTPSConnection("account.ypw.com.do")

payload = json.dumps({
    "username": "null",
    "password": "null",
    "appConnect": "miapp"
})

headers = {
    "Content-Type": "application/json"
}

conn.request("POST", "/api/v1/account/login", payload, headers)
res = conn.getresponse()
data = res.read()

print(data.decode("utf-8"))
